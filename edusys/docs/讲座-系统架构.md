## 第一部分：系统概览

### 1.1 项目背景

EduSys 是一个面向中小学的综合性学校管理系统，涵盖以下核心模块：

- **用户与权限管理** - 基于 RBAC 模型的权限控制
- **教务管理** - 年级、班级、学生、教师、科目管理
- **考试管理** - 考试安排、场次、考点、考场编排
- **成绩管理** - 成绩导入、解析、学生匹配、正式表导入

### 1.2 技术栈

| 层级 | 技术选型 | 版本 |
|------|----------|------|
| 前端 | Vue 3 + TypeScript + Element Plus | Vue 3.4+ |
| 后端 | NestJS + TypeORM | NestJS 10+ |
| 数据库 | PostgreSQL | 15+ |
| Python服务 | FastAPI + Pandas | Python 3.10+ |
| 部署 | Docker + Docker Compose | 24+ |

### 1.3 项目规模

```
┌─────────────┐    ┌─────────────┐    ┌─────────────┐    ┌─────────────┐
│  选择考试    │ -> │  选择系统   │ -> │  上传文件   │ -> │  Python解析  │
│  (新建考试)  │    │ (好分数/    │    │  (.xlsx)    │    │  Excel解析   │

└─────────────┘    └─────────────┘    └─────────────┘    └─────────────┘
                                                              │
                                                              ▼

                   │  导入正式表  │ <- │  学生匹配   │ <- │  解析完成   │
                   │  (事务)     │    │  (多策略)  │    │  返回数据   │
                   └─────────────┘    └─────────────┘    └─────────────┘
```

**详细流程：**
1. **选择考试** - 新建或选择已有考试，关联年级
2. **选择系统** - 好分数/睿芽/七天/自定义
3. **上传文件** - 支持 .xlsx/.xls，最大10MB
4. **Python解析** - 识别表头、提取学生信息、解析各科成绩
5. **学生匹配** - 考号→学籍号→班级+姓名，优先级匹配
6. **保存正式表** - 事务写入，覆盖模式

### 4.4 Python 微服务

#### 4.4.4 学生匹配器实现

```python
# app/services/matcher.py
class StudentMatcher:
    """学生匹配器 - 多策略匹配"""

    def __init__(self, existing_students: List[Dict]):
        self.existing_students = existing_students
        self._build_indexes()  # 构建索引加速匹配

    def _build_indexes(self):
        """构建索引"""
        for student in self.existing_students:
            # 考号索引
            if student.get('studentNo'):
                self._add_to_index('student_no', student['studentNo'], student)
            # 学籍号索引（去除H后）
            if student.get('studentId'):
                sid = student['studentId'].replace('H', '')
                self._add_to_index('student_id_no_h', sid, student)
            # 班级+姓名索引
            if student.get('className') and student.get('name'):
                key = f"{student['className']}|{student['name']}"
                self._add_to_index('class_name', key, student)

    def match(self, student: StudentScore) -> StudentScore:
        """匹配单个学生 - 优先级策略"""
        # 策略1: 考号精确匹配
        matched, method = self._match_by_student_no(student)
        if matched:
            return self._set_match_result(student, matched, method)

        # 策略2: 学籍号匹配（去除H）
        matched, method = self._match_by_student_id_no_h(student)
        if matched:
            return self._set_match_result(student, matched, method)

        # 策略3: 班级+姓名匹配
        matched, method = self._match_by_class_name(student)
        if matched:
            return self._set_match_result(student, matched, method)

        # 未匹配
        student.match_status = 'unmatched'
        return student
```

```typescript
// stores/user.ts - 用户状态管理
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'

export const useUserStore = defineStore('user', () => {
  const userInfo = ref<any>(null)
  const permissions = ref<string[]>([])
  const isSuperAdmin = computed(() => userInfo.value?.role?.isSuperAdmin === 1)

  // 检查是否有权限
  const hasPermission = (permissionCode: string): boolean => {
    // 超级管理员拥有所有权限
    if (isSuperAdmin.value) return true
    // 检查权限列表
    return permissions.value.includes(permissionCode)
  }

  // 从服务器加载用户信息和权限
  const getUserInfo = async () => {
    const res = await getUserProfile()
    userInfo.value = res.user
    permissions.value = res.permissions || []
  }

  return { userInfo, permissions, isSuperAdmin, hasPermission, getUserInfo }
})

// 路由守卫中的使用
router.beforeEach((to, from) => {
```

```typescript
// score-import.controller.ts
import { Controller, Post, Body, UseInterceptors, UploadedFile, Req } from '@nestjs/common'
import { FileInterceptor } from '@nestjs/platform-express'
import type { Request } from 'express'

@Controller('score-import')
export class ScoreImportController {
  constructor(
    @InjectRepository(Student)
    private studentRepository: Repository<Student>,
    // ...其他Repository
  ) {}

  /**
   * 解析Excel文件（调用Python微服务）
   */
  @Post('parse')
  @UseInterceptors(FileInterceptor('file'))
  async parseExcel(
    @UploadedFile() file: any,
    @Body() body: { system?: string }
  ) {
    try {
      // 将文件转发到 Python 微服务
      const formData = new FormData()
      formData.append('file', file.buffer, {
        filename: file.originalname,
        contentType: file.mimetype
      })

      // 如果指定了系统，添加到请求中
      if (body?.system) {
        formData.append('system', body.system)
      }

      const response = await axios.post(
        `${PYTHON_API}/python/score/parse`,
        formData,
        {
          headers: {
            ...formData.getHeaders()
          }
        }
      )

      return response.data
    } catch (error: any) {
      console.error('解析文件失败:', error.message)
      return {
        success: false,
        message: '解析文件失败: ' + error.message
      }
    }
  }
}
```

```typescript
async saveToExamScore(@Body() body: { examId: number }) {
  const { examId } = body

  // 创建查询运行器
  const queryRunner = this.tempRepository.manager.connection.createQueryRunner()
  await queryRunner.connect()
  await queryRunner.startTransaction()

      [examId]
    )
    console.log(`已删除考试ID=${examId}的原有成绩`)

    // 2. 查询临时表中未确认的数据
    const tempData = await this.tempRepository.find({
      where: { examId, status: 0 },
      relations: ['student']
    })

    let savedCount = 0

    // 3. 遍历临时数据，插入正式表
    for (const temp of tempData) {
      // 跳过没有匹配到学生的记录
      if (!temp.studentId) continue

      // 验证学生ID是否存在
      const studentExists = await queryRunner.query(
        'SELECT id FROM student WHERE id = $1',
        [temp.studentId]
      )
      if (studentExists.length === 0) continue

      // 插入到 exam_score 表
      await queryRunner.query(
        `INSERT INTO exam_score (
          exam_id, student_id, import_batch,
          chinese, chinese_rank, math, math_rank, ...
          total_score, total_rank, status, matched_method
        ) VALUES ($1, $2, $3, $4, $5, ...)`,
        [
          examId,
          temp.studentId,
          temp.importBatch || `EXAM_${examId}_${Date.now()}`,
          temp.chinese, temp.chineseRank, temp.math, temp.mathRank,
          temp.totalScore, temp.totalRank,
          1, // status: 已确认
          temp.matchedMethod || '自动匹配'
        ]
      )
      savedCount++

      // 4. 更新临时表状态为已导入 (status = 2)
      await queryRunner.query(
        'UPDATE score_import_temp SET status = 2 WHERE id = $1',
        [temp.id]
      )
    }

    // 5. 更新考试的 has_scores 标记
    await queryRunner.query(
      `UPDATE exam SET has_scores = 1, score_import_time = NOW() WHERE id = $1`,
      [examId]
    )

    // 6. 提交事务
    await queryRunner.commitTransaction()

    return {
      success: true,
      message: `成功导入 ${savedCount} 条成绩`,
      count: savedCount
    }
  } catch (error: any) {
    // 回滚事务
    await queryRunner.rollbackTransaction()
    console.error('保存原始成绩失败:', error)
    return { success: false, message: '保存失败: ' + error.message }
  } finally {
    // 释放连接
    await queryRunner.release()
  }
}
```

#### 5.3.2 事务的必要性

| 场景 | 无事务 | 有事务 |
|------|--------|--------|
| 插入10条，第5条失败 | 前4条已入库，后5条丢失 | 全部回滚，数据一致性 |
| 程序中断 | 部分数据写入，导致数据不完整 | 全部回滚 |
| 并发导入 | 数据覆盖冲突 | 锁机制，保证一致性 |

### 5.4 操作日志记录

```typescript
// interceptors/operation-log.interceptor.ts
@Injectable()
export class OperationLogInterceptor implements NestInterceptor {
  constructor(private dataSource: DataSource) {}

  intercept(context: ExecutionContext, next: CallHandler): Observable<any> {
    const request = context.switchToHttp().getRequest()
    const response = context.switchToHttp().getResponse()

    // 只记录POST/PUT/DELETE请求
    if (!['POST', 'PUT', 'DELETE'].includes(request.method)) {
      return next.handle()
    }

    // 深拷贝body，避免被修改
    const body = JSON.parse(JSON.stringify(request.body))

    return next.handle().pipe(
      tap({
        next: (data) => {
          // 请求成功后记录日志
          this.saveLog({
            userId: request.user?.id,
            username: request.user?.username,
            method: request.method,
            path: request.path,
            body: body,
            response: data,
            status: response.statusCode,
            ip: request.ip
          })
        },
        error: (error) => {
          // 请求失败后记录日志
          this.saveLog({
            userId: request.user?.id,
            username: request.user?.username,
            method: request.method,
            path: request.path,
            body: body,
            error: error.message,
            status: response.statusCode,
            ip: request.ip
          })
        }
      })
    )
  }

  private async saveLog(logData: any) {
    const repository = this.dataSource.getRepository(OperationLog)
    const log = repository.create(logData)
    await repository.save(log)
  }
}
```

## 第八部分：实战案例 - 成绩导入模块

### 8.1 需求背景

学校需要从不同考试系统导入成绩，数据格式差异大：
- **好分数**：多行表头，列位置固定
- **睿芽**：单行表头，列位置不固定
- **七天**：选科组合，赋分制
- **自定义**：学校自己的格式

### 8.2 技术挑战

1. **格式多样** - 每种系统解析逻辑不同
2. **学生匹配** - 考号、学籍号、姓名都可能不一致
3. **数据量大** - 单次导入可达数千条
4. **事务安全** - 导入失败需回滚

### 8.3 解决方案

| 挑战 | 解决方案 |
|------|----------|
| 格式多样 | Python + Pandas 解析，根据系统类型选择解析策略 |
| 学生匹配 | 多策略优先级匹配：考号→学籍号→班级+姓名 |
| 数据量大 | 分批处理，进度条展示 |
| 事务安全 | QueryRunner 事务管理 |

## 第十一部分：Q&A

### 常见问题

**Q: 为什么不直接用后端解析 Excel？**
A: Python 的 Pandas 库对 Excel 处理更成熟，不同系统的格式差异大，Python 更适合做数据处理。

**Q: 如何保证学生匹配的准确性？**
A: 采用多策略优先级匹配：考号精确匹配 → 学籍号匹配 → 班级+姓名匹配，匹配率可达 95%+。

**Q: 如何处理大规模数据导入？**
A: 使用事务保证一致性，分批处理并显示进度条，前端有进度提示。

**Q: 如何实现权限控制？**
A: RBAC 模型，用户关联角色，角色关联权限，菜单关联权限码，路由守卫检查权限。
